## AI Data Retrieval and Ranking Query Design

# Create the Table:
CREATE TABLE STUDENTS (
id Integer Primary Key,
name Text,
score Integer,
passed Boolean,
class Integer
);

# Retrieve all columns from the students table:
SELECT * FROM STUDENTS;

# Retrieve only the name and score columns:
SELECT name, score FROM STUDENTS;

# Retrieve students who scored above 80:
SELECT * FROM STUDENTS WHERE score > 80;

# Retrieve students who scored above 80 AND passed.
SELECT * FROM STUDENTS WHERE score > 80 AND passed = TRUE;

# Retrieve students who scored above 85 OR belong to class 9
SELECT * FROM STUDENTS WHERE score > 85 OR class = 9;

# Retrieve students who did NOT pass the exam
SELECT * FROM STUDENTS WHERE NOT passed;
